// Imports
const {  } = require('./index.js');

// Main
class URLParams {
    /**
     * Creates a new set of URLEncoded parameters.
     * @constructor
     * @param {*} data The data to use
     */
    constructor(data) {
        this.data = data;
        this.string = null;
        this._validateType(data);
        this._init();
    }

    /**
     * Validates the type of data
     * @function
     * @param {*} data The data
     * @private
     */
    _validateType(data) {
        let s = this._checkType(data);
        if (s === null) {
            // no data
            throw new Err('URLParams takes 1 argument: provided 0', 'ArgumentCountError');
        } else if (s === false) {
            // bad type
            throw new Err('URLParams was passed an invalid type: expected Object', 'TypeError');
        } else {
            // ok, we return
            return;
        }
    }

    /**
     * Checks the type of data
     * @function
     * @param {*} data The data
     * @private
     */
    _checkType(data) {
        if (!data) {
            return null;
        } else if (typeof data !== 'object') {
            return false;
        } else {
            return true;
        }
    }

    /**
     * Initializes the parser
     * @function
     * @private
     */
    _init() {
        let d = this.data;
        d = Object.entries(d);
        if (d.length === 0) {
            this.string = '';
        } else {
            for (let i of d) {
                this._appendString(i[0], i[1]);
            }
        }
    }

    /**
     * Appends to the query string.
     * @function
     * @param {*} key The key
     * @param {*} val The value
     * @private
     */
    _appendString(key, val) {
        if (this.string) {
            this.string += `&${this._urlEncode(key)}=${this._urlEncode(val)}`
        } else {
            this.string = `${key}=${val}`
        }
    }

    /**
     * Encodes the URI component.
     * @function
     * @param {*} data The data to encode
     * @private
     */
    _urlEncode(data) {
        let out = data;

        // Parse data types
        let t = typeof out;
        switch (t) {
            case 'string':
                break;
            case 'boolean':
                out = out.toString();
                break;
            case 'object':
                out = JSON.stringify(out);
                break;
            default:
                break;
        }

        // Parse URLEncoded stuff
        out = encodeURIComponent(out);

        return out;
    }
}

// Export
module.exports = URLParams;
