// Imports
const {  } = require('./index.js');

// Main
/**
 * A custom error with message
 * @extends Error
 */
class Err extends Error {
    /**
     * Creates a new custom error (With message)
     * @constructor
     * @param {String} message The error message
     * @param {String} [type] The type of error (Like "TypeError")
     * @example
     * throw new Err('Example', 'ExampleError'); // => [NodeVTError]: ExampleError: Example
     */
    constructor(message, type) {
        super(`${type ? type + ': ' : ''}${message}`);
        this.name = '[NodeVTError]'
    }
}

// Export
module.exports = Err;
